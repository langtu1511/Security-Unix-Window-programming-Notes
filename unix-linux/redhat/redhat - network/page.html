[+] Configure bridge
  • cat /etc/sysconfig/network-scripts/ifcfg-br0
    • DEVICE=br0
    • TYPE=Dridge
    • BOOTPROTO=static
    • ONBOOT=yes
    • IPADDR=192.168.0.1
    • NETMASK=255.255.255.0
    • GATEWAY=192.168.0.254
    • NM_CONTROLLED=no
  • cat /etc/sysconfig/network-scripts/ifcfg-eth0
    • DEVICE=eth0
    • TYPE=Ethernet
    • ONBOOT=yes
    • BRIDGE=br0
    • NM_CONTROLLED=no





END
